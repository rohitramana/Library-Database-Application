create table BOOK   ( ISBN   varchar(13)    NOT NULL UNIQUE, 	TITLE  varchar(300) CHARACTER SET UTF8, COVER  varchar(300), PUBLISHER varchar(100) CHARACTER SET UTF8, PAGES INT DEFAULT 0, CONSTRAINT ISBNPK PRIMARY KEY (ISBN));

create table BORROWER ( CARD_ID varchar(14) not null unique , SSN VARCHAR(15) UNIQUE not NULL, B_FIRST_NAME VARCHAR(13), B_LAST_NAME VARCHAR(13), EMAIL VARCHAR(50) UNIQUE, ADDRESS VARCHAR(100), CITY VARCHAR(20), STATE VARCHAR(20), PHONE VARCHAR(14), CONSTRAINT PRIMARY KEY (CARD_ID))

create table AUTHORS ( AUTHOR_ID INT AUTO_INCREMENT, AUTHOR_NAME varchar(100) CHARACTER SET UTF8 UNIQUE , CONSTRAINT AUTHOR_IDPK PRIMARY KEY (AUTHOR_ID));

create table BOOK_LOANS ( Loan_id INT auto_increment not null unique, ISBN  VARCHAR(13), CARD_ID varchar(14), DATE_OUT DATE, DUE_DATE DATE, DATE_IN DATE, CONSTRAINT FOREIGN KEY (ISBN)  REFERENCES BOOK(ISBN), CONSTRAINT FOREIGN KEY(CARD_ID) REFERENCES BORROWER(CARD_ID) on delete cascade) 

create table BOOK_AUTHOR ( AUTHOR_ID INT, ISBN varchar(13), CONSTRAINT AUTHOR_IDFK FOREIGN KEY (AUTHOR_ID) REFERENCES AUTHORS(AUTHOR_ID), CONSTRAINT ISBNFK FOREIGN KEY (ISBN) REFERENCES BOOK(ISBN) on delete cascade);

create table FINES ( LOAN_ID INT, FINE_AMT FLOAT, PAID BOOLEAN, constraint foreign key (loan_id) references book_loans(loan_id))

create table BOOK_AVAIL ( ISBN varchar(13) not null unique, availability int default =1, constraint foreign key (ISBN) references book(ISBN) on delete cascade)
